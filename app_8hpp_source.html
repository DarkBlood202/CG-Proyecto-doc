187 void AppOpenGL::installLights(glm::mat4 vMatrix)
188 {
189  // convert lightÔøΩs position to view space, and save it in a float array
190  lightPosV = glm::vec3(vMatrix * glm::vec4(currentLightPos, 1.0));
191  lightPos[0] = lightPosV.x;
192  lightPos[1] = lightPosV.y;
193  lightPos[2] = lightPosV.z;
194  // get the locations of the light and material fields in the shader
195  globalAmbLoc = glGetUniformLocation(sp.getID(), "globalAmbient");
196  ambLoc = glGetUniformLocation(sp.getID(), "light.ambient");
197  diffLoc = glGetUniformLocation(sp.getID(), "light.diffuse");
198  specLoc = glGetUniformLocation(sp.getID(), "light.specular");
199  posLoc = glGetUniformLocation(sp.getID(), "light.position");
200  mAmbLoc = glGetUniformLocation(sp.getID(), "material.ambient");
201  mDiffLoc = glGetUniformLocation(sp.getID(), "material.diffuse");
202  mSpecLoc = glGetUniformLocation(sp.getID(), "material.specular");
203  mShiLoc = glGetUniformLocation(sp.getID(), "material.shininess");
204  // set the uniform light and material values in the shader
205  glProgramUniform4fv(sp.getID(), globalAmbLoc, 1, globalAmbient);
206  glProgramUniform4fv(sp.getID(), ambLoc, 1, lightAmbient);
207  glProgramUniform4fv(sp.getID(), diffLoc, 1, lightDiffuse);
208  glProgramUniform4fv(sp.getID(), specLoc, 1, lightSpecular);
209  glProgramUniform3fv(sp.getID(), posLoc, 1, lightPos);
210  glProgramUniform4fv(sp.getID(), mAmbLoc, 1, matAmb);
211  glProgramUniform4fv(sp.getID(), mDiffLoc, 1, matDif);
212  glProgramUniform4fv(sp.getID(), mSpecLoc, 1, matSpe);
213  glProgramUniform1f(sp.getID(), mShiLoc, matShi);
214 }
