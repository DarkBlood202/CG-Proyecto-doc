90  // y = r * cos(u) * sin(v)
91  // z = r * sin(u)
92  // where u: stack(latitude) angle (-90 <= u <= 90)
93  // v: sector(longitude) angle (0 <= v <= 360)
94 
95  const float PI = acos(-1);
96 
97  // clear memory of prev arrays
98  clearArrays();

100  float x, y, z, xy; // vertex position
101  float nx, ny, nz, lengthInv = 1.0f / radius; // normal
102  float s, t; // texCoord
103 
104  float sectorStep = 2 * PI / sectorCount;
105  float stackStep = PI / stackCount;
106  float sectorAngle, stackAngle;
107 
108  for(int i = 0; i <= stackCount; ++i)
109  {
110  stackAngle = PI / 2 - i * stackStep; // starting from pi/2 to -pi/2
111  xy = radius * cosf(stackAngle); // r * cos(u)
112  z = radius * sinf(stackAngle); // r * sin(u)
113 
114  // add (sectorCount+1) vertices per stack
115  // the first and last vertices have same position and normal, but different tex coords
116  for(int j = 0; j <= sectorCount; ++j)
117  {
118  sectorAngle = j * sectorStep; // starting from 0 to 2pi
119 
120  // vertex position
121  x = xy * cosf(sectorAngle); // r * cos(u) * cos(v)
122  y = xy * sinf(sectorAngle); // r * cos(u) * sin(v)
123  addVertex(x, y, z);
124 
125  // normalized vertex normal
126  nx = x * lengthInv;
127  ny = y * lengthInv;
128  nz = z * lengthInv;
129  addNormal(nx, ny, nz);
130 
131  // vertex tex coord between [0, 1]
132  s = (float)j / sectorCount;
133  t = (float)i / stackCount;
134  addTexCoord(s, t);
135  }
136  }
137 
138  // indices
139  // k1--k1+1
140  // | / |
141  // | / |
142  // k2--k2+1
143  unsigned int k1, k2;
144  for(int i = 0; i < stackCount; ++i) {
145  k1 = i * (sectorCount + 1); // beginning of current stack
146  k2 = k1 + sectorCount + 1; // beginning of next stack
147 
148  for(int j = 0; j < sectorCount; ++j, ++k1, ++k2) {
149  // 2 triangles per sector excluding 1st and last stacks
150  if(i != 0)

153  }
154 
155  if(i != (stackCount-1))

158  }
159

162  lineIndices.push_back(k2);
163  if(i != 0) // horizontal lines except 1st stack

167  }
168  }
169  }
