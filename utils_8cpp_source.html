59 GLuint Utils::prepareShader(int shaderTYPE, const char *shaderPath)
60 {
61  GLint shaderCompiled;
62  string shaderStr = readShaderFile(shaderPath);
63  const char *shaderSrc = shaderStr.c_str();
64  GLuint shaderRef = glCreateShader(shaderTYPE);
65  glShaderSource(shaderRef, 1, &shaderSrc, NULL);
66  glCompileShader(shaderRef);
67  checkOpenGLError();
68  glGetShaderiv(shaderRef, GL_COMPILE_STATUS, &shaderCompiled);
69  if (shaderCompiled != 1)
70  {
71  if (shaderTYPE == 35633) cout << "Vertex ";
72  if (shaderTYPE == 36488) cout << "Tess Control ";
73  if (shaderTYPE == 36487) cout << "Tess Eval ";
74  if (shaderTYPE == 36313) cout << "Geometry ";
75  if (shaderTYPE == 35632) cout << "Fragment ";
76  cout << "shader compilation error." << endl;
77  printShaderLog(shaderRef);
78  }
79  return shaderRef;
80 }
