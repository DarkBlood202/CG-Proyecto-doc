165 void AppOpenGL::installLights(glm::mat4 vMatrix)
166 {
167  // convert light�s position to view space, and save it in a float array
168  lightPosV = glm::vec3(vMatrix * glm::vec4(currentLightPos, 1.0));
169  lightPos[0] = lightPosV.x;
170  lightPos[1] = lightPosV.y;
171  lightPos[2] = lightPosV.z;
172  // get the locations of the light and material fields in the shader
173  globalAmbLoc = glGetUniformLocation(sp.getID(), "globalAmbient");
174  ambLoc = glGetUniformLocation(sp.getID(), "light.ambient");
175  diffLoc = glGetUniformLocation(sp.getID(), "light.diffuse");
176  specLoc = glGetUniformLocation(sp.getID(), "light.specular");
177  posLoc = glGetUniformLocation(sp.getID(), "light.position");
178  mAmbLoc = glGetUniformLocation(sp.getID(), "material.ambient");
179  mDiffLoc = glGetUniformLocation(sp.getID(), "material.diffuse");
180  mSpecLoc = glGetUniformLocation(sp.getID(), "material.specular");
181  mShiLoc = glGetUniformLocation(sp.getID(), "material.shininess");
182  // set the uniform light and material values in the shader
183  glProgramUniform4fv(sp.getID(), globalAmbLoc, 1, globalAmbient);
184  glProgramUniform4fv(sp.getID(), ambLoc, 1, lightAmbient);
185  glProgramUniform4fv(sp.getID(), diffLoc, 1, lightDiffuse);
186  glProgramUniform4fv(sp.getID(), specLoc, 1, lightSpecular);
187  glProgramUniform3fv(sp.getID(), posLoc, 1, lightPos);
188  glProgramUniform4fv(sp.getID(), mAmbLoc, 1, matAmb);
189  glProgramUniform4fv(sp.getID(), mDiffLoc, 1, matDif);
190  glProgramUniform4fv(sp.getID(), mSpecLoc, 1, matSpe);
191  glProgramUniform1f(sp.getID(), mShiLoc, matShi);
192 }
