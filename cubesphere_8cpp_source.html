364 // return face normal (4th param) of a triangle v1-v2-v3
365 // if a triangle has no surface (normal length = 0), then return a zero vector
367 void Cubesphere::computeFaceNormal(const float v1[3], const float v2[3], const float v3[3], float n[3])
368 {
369  const float EPSILON = 0.000001f;
370 
371  // default return value (0, 0, 0)
372  n[0] = n[1] = n[2] = 0;
373 
374  // find 2 edge vectors: v1-v2, v1-v3
375  float ex1 = v2[0] - v1[0];
376  float ey1 = v2[1] - v1[1];
377  float ez1 = v2[2] - v1[2];
378  float ex2 = v3[0] - v1[0];
379  float ey2 = v3[1] - v1[1];
380  float ez2 = v3[2] - v1[2];
381 
382  // cross product: e1 x e2
383  float nx, ny, nz;
384  nx = ey1 * ez2 - ez1 * ey2;
385  ny = ez1 * ex2 - ex1 * ez2;
386  nz = ex1 * ey2 - ey1 * ex2;
387 
388  // normalize only if the length is > 0
389  float length = sqrtf(nx * nx + ny * ny + nz * nz);
390  if(length > EPSILON)
391  {
392  // normalize
393  float lengthInv = 1.0f / length;
394  n[0] = nx * lengthInv;
395  n[1] = ny * lengthInv;
396  n[2] = nz * lengthInv;
397  }
398 }

414 // generate vertices for +X face only by intersecting 2 circular planes
415 // (longitudinal and latitudinal) at the longitude/latitude angles
417 std::vector<float> Cubesphere::getUnitPositiveX(unsigned int pointsPerRow)
418 {
419  const float DEG2RAD = acos(-1) / 180.0f;
420 
421  std::vector<float> vertices;
422  float n1[3]; // normal of longitudinal plane rotating along Y-axis
423  float n2[3]; // normal of latitudinal plane rotating along Z-axis
424  float v[3]; // direction vector intersecting 2 planes, n1 x n2
425  float a1; // longitudinal angle along y-axis
426  float a2; // latitudinal angle
427  float scale;
428 
429  // rotate latitudinal plane from 45 to -45 degrees along Z-axis
430  for(unsigned int i = 0; i < pointsPerRow; ++i)
431  {
432  // normal for latitudinal plane
433  a2 = DEG2RAD * (45.0f - 90.0f * i / (pointsPerRow - 1));
434  n2[0] = -sin(a2);
435  n2[1] = cos(a2);
436  n2[2] = 0;
437 
438  // rotate longitudinal plane from -45 to 45 along Y-axis
439  for(unsigned int j = 0; j < pointsPerRow; ++j)
440  {
441  // normal for longitudinal plane
442  a1 = DEG2RAD * (-45.0f + 90.0f * j / (pointsPerRow - 1));
443  n1[0] = -sin(a1);
444  n1[1] = 0;
445  n1[2] = -cos(a1);
446 
447  // find direction vector of intersected line, n1 x n2
448  v[0] = n1[1] * n2[2] - n1[2] * n2[1];
449  v[1] = n1[2] * n2[0] - n1[0] * n2[2];
450  v[2] = n1[0] * n2[1] - n1[1] * n2[0];
451 
452  // normalize direction vector
453  scale = Cubesphere::computeScaleForLength(v, 1);
454  v[0] *= scale;
455  v[1] *= scale;
456  v[2] *= scale;
457 
458  vertices.push_back(v[0]);
459  vertices.push_back(v[1]);
460  vertices.push_back(v[2]);
461 
462  // DEBUG
463  //std::cout << "vertex: (" << v[0] << ", " << v[1] << ", " << v[2] << "), "
464  // << sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]) << std::endl;
465  }
466  }
467 
468  return vertices;
469 }
